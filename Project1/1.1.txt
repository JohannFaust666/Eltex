#include <stdio.h>
#include <sys/types.h>
#include <unistd.h>
#include <limits.h>

int main() {
    // Возвращает идентификатор текущего процесса
    pid_t pid = getpid();
    printf("My PID is %d\n", pid);
    
    // Возвращает имя текущего хоста
    char hostname[HOST_NAME_MAX];
    gethostname(hostname, HOST_NAME_MAX);
    printf("My hostname is %s\n", hostname);
    
    // Возвращает текущий рабочий каталог
    char cwd[PATH_MAX];
    getcwd(cwd, sizeof(cwd));
    printf("My current working directory is %s\n", cwd);
    
    // Возвращает идентификатор пользователя, от имени которого запущена программа
    uid_t uid = getuid();
    printf("My UID is %d\n", uid);
    
    // Возвращает идентификатор группы, от имени которой запущена программа
    gid_t gid = getgid();
    printf("My GID is %d\n", gid);
    
    return 0;
}
